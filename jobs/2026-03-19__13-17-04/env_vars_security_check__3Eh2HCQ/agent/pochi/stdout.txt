import json
from e2b import Sandbox

def main():
    s = Sandbox.create(
        envs={
            "SECRET_TOKEN": "super-secret-123",
            "API_ENDPOINT": "https://api.example.com"
        },
        timeout=3600 # Keep alive for 1 hour
    )
    
    cmd = 'echo "TOKEN=$SECRET_TOKEN" > /home/user/env_dump.txt && echo "ENDPOINT=$API_ENDPOINT" >> /home/user/env_dump.txt'
    s.commands.run(cmd)
    
    with open('/home/user/e2b_task_info.json', 'w') as f:
        json.dump({"sandbox_id": s.sandbox_id}, f)
        
    print(f"Sandbox ID: {s.sandbox_id}")

if __name__ == "__main__":
    main()
