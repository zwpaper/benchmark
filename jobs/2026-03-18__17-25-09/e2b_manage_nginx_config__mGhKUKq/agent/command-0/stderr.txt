4.  **Verifies Content**: Reads the file back from the sandbox filesystem to ensure the content was written correctly.

### Note:
To run the script, ensure the `e2b` package is installed (`pip install e2b`) and the `E2B_API_KEY` environment variable is configured.

```python
import json
import os
from e2b import Sandbox

def main():
    # Content of the Nginx configuration file
    nginx_conf = """server {
    listen 8080;
    server_name localhost;
    location / {
        root /var/www/html;
        index index.html;
    }
}"""

    try:
        # 1. Create a new sandbox environment
        print("Creating a new sandbox...")
        sb = Sandbox()
        print(f"Sandbox created with ID: {sb.id}")

        # 2. Create a directory /etc/custom_nginx/ inside the sandbox
        print("Creating directory /etc/custom_nginx/...")
        sb.filesystem.make_dir("/etc/custom_nginx/")

        # 3. Write configuration file /etc/custom_nginx/nginx.conf
        print("Writing /etc/custom_nginx/nginx.conf...")
        sb.filesystem.write("/etc/custom_nginx/nginx.conf", nginx_conf)

        # 4. Read it back to verify the content
        print("Reading back the file to verify content...")
        read_content = sb.filesystem.read("/etc/custom_nginx/nginx.conf")
        
        if read_content == nginx_conf:
            print("Verification successful: Content matches.")
        else:
            print("Verification failed: Content does not match.")

        # 5. Keep the sandbox alive by setting a long timeout (1 hour)
        sb.set_timeout(3600)
        print("Sandbox timeout set to 3600 seconds to keep it alive.")

        # 6. Write a JSON file to /home/user/e2b_task_info.json containing the sandbox_id
        host_json_path = "/home/user/e2b_task_info.json"
        with open(host_json_path, "w") as f:
            json.dump({"sandbox_id": sb.id}, f, indent=4)
        print(f"Sandbox info written to {host_json_path}")

    except Exception as e:
        print(f"An error occurred: {e}")

if __name__ == "__main__":
    main()
```   
