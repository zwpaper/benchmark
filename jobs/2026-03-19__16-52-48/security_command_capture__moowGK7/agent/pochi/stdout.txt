import json
from e2b import Sandbox

sandbox = Sandbox.create(timeout=600)

script_content = """#!/bin/bash
echo 'Scanning open ports...'
echo 'Found vulnerable port 8080' >&2
echo 'Scan complete. 1 vulnerabilities found.'
"""

sandbox.files.write("/home/user/scan_network.sh", script_content)
res = sandbox.commands.run("chmod +x /home/user/scan_network.sh")
print('chmod', res.exit_code, res.stderr)

result = sandbox.commands.run("/home/user/scan_network.sh")
print('run', result.exit_code, result.stdout, result.stderr)

output_content = f"STDOUT:\n{result.stdout}\nSTDERR:\n{result.stderr}"
sandbox.files.write("/home/user/scan_results.txt", output_content)

with open("/home/user/e2b_task_info.json", "w", encoding="utf-8") as file:
    json.dump({"sandbox_id": sandbox.id}, file)
print('done')
