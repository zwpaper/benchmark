import json
from e2b import Sandbox

def main():
    metadata = {
        "user_id": "devops_user_99",
        "environment": "staging",
        "purpose": "infrastructure_testing"
    }
    
    # 1 & 2. Create sandbox with metadata
    sandbox = Sandbox.create(metadata=metadata)
    
    # 3. Write sandbox ID to file
    with open('/home/user/sandbox_id.txt', 'w') as f:
        f.write(sandbox.sandbox_id)
        
    # 4. Get list of running sandboxes
    running_sandboxes = Sandbox.list()
    
    # 5. Find the sandbox
    target_sandbox = None
    while True:
        items = running_sandboxes.next_items()
        for s in items:
            if s.sandbox_id == sandbox.sandbox_id:
                target_sandbox = s
                break
        if target_sandbox or not running_sandboxes.has_next:
            break
            
    # 6. Write retrieved metadata to json
    if target_sandbox:
        with open('/home/user/retrieved_metadata.json', 'w') as f:
            json.dump(target_sandbox.metadata, f)
            
    # 7. Leave sandbox running
    sandbox.set_timeout(3600)

if __name__ == "__main__":
    main()
