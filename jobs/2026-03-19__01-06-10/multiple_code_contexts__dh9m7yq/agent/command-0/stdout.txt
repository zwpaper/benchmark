```python
import os
from e2b_code_interpreter import Sandbox

def main():
    # 1. Connects to a single E2B Sandbox using the e2b_code_interpreter SDK.
    # We use a context manager to ensure the sandbox is closed at the end.
    print("Creating sandbox...")
    with Sandbox.create() as sandbox:
        # 2. Creates two separate code execution contexts within this sandbox
        print("Creating execution contexts...")
        ctx1 = sandbox.create_code_context()
        ctx2 = sandbox.create_code_context()

        # 3. In ctx1, execute Python code that sets a variable magic_number = 42
        print("Setting magic_number in ctx1...")
        sandbox.run_code("magic_number = 42", context=ctx1)

        # 4. In ctx2, execute Python code that sets a variable magic_number = 99
        print("Setting magic_number in ctx2...")
        sandbox.run_code("magic_number = 99", context=ctx2)

        # 5. In ctx1, execute Python code to read magic_number and write it to a file /home/user/out1.txt
        print("Writing out1.txt in ctx1...")
        sandbox.run_code(
            "with open('/home/user/out1.txt', 'w') as f: f.write(str(magic_number))",
            context=ctx1
        )

        # 6. In ctx2, execute Python code to read magic_number and write it to a file /home/user/out2.txt
        print("Writing out2.txt in ctx2...")
        sandbox.run_code(
            "with open('/home/user/out2.txt', 'w') as f: f.write(str(magic_number))",
            context=ctx2
        )

        # 7. Finally, download both files from the sandbox to the local workspace
        print("Downloading files...")
        
        # sandbox.files.read returns the file content. 
        # By default, it returns a string for text files.
        
        out1_content = sandbox.files.read("/home/user/out1.txt")
        with open("host_out1.txt", "w") as f:
            f.write(out1_content)
        print("Saved host_out1.txt")

        out2_content = sandbox.files.read("/home/user/out2.txt")
        with open("host_out2.txt", "w") as f:
            f.write(out2_content)
        print("Saved host_out2.txt")

    # 8. Ensure the sandbox is closed at the end.
    # Handled by the 'with' statement.
    print("Sandbox closed.")

if __name__ == "__main__":
    # Note: E2B_API_KEY must be set in the environment for this to run.
    if not os.environ.get("E2B_API_KEY"):
        print("Warning: E2B_API_KEY environment variable is not set.")
    main()
```
